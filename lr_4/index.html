Лабораторная работа № 4

Описание: Необходимо упаковать ваше приложение в docker-контейнеры и обеспечить сетевое взаимодействие между различными частями вашего приложения. Делать это можно как с помощью docker-compose так и с помощью docker swarm. При разумном использовании swirl вы получите дополнительные баллы.

Ход работы

Dockerfile для gateway 

FROM node:16-alpine

WORKDIR /app

COPY package*.json ./

COPY . . 

RUN npm install

EXPOSE 3000

CMD ["make", "start"]

Dockerfile для main_app

FROM node:16-alpine

WORKDIR /app

COPY package*.json ./

COPY . . 

RUN npm install

EXPOSE 9090

CMD ["make", "start"]

Dockerfile для microservice

FROM node:16-alpine

WORKDIR /app

COPY package*.json ./

COPY . . 

RUN npm install

EXPOSE 9091

CMD ["make", "start"]

Docker compose

version: '3.8'

services:
  gateway:
    container_name: gateway
    build:
      context: ./gateway
    volumes:
      - ./gateway:/app
      - /app/node_modules
    command: npx nodemon index.js
    ports:
      - 3000:3000
    restart: always
  main_app:
    container_name: main_app
    build:
      context: ./main_app
    volumes:
      - ./main_app:/app
      - /app/node_modules
    depends_on:
      - microservice
    command: npx nodemon dist/index.js
    ports:
      - 9090:9090
    restart: always
  microservice:
    container_name: microservice
    build:
      context: ./microservice
    depends_on:
      - gateway
    command: npx nodemon dist/index.js
    volumes:
      - ./microservice:/app
      - /app/node_modules
    ports:
      - 9091:9091 
    restart: always

Вывод

В ходе данной работы я упаковал приложение в docker контейнер и обеспечил взаимодействие между различными частями приложения с помощью docker compose.
